
// Elabore um algoritmo que calcule e mostre a soma e a média dos números pares entre 1 e 100, inclusive. Apresente a representação em Pseudocódigo.

programa {
  funcao inicio() {
    
    inteiro soma, quantidade, numero, media
    
    // Inicialização das variáveis
    soma = 0
    quantidade = 0
    
    // Percorrer números de 1 a 100
    para(numero = 1; numero <= 100; numero++){
      
      // Verificar se é par
      se (numero % 2 == 0) {
        soma = soma + numero      // Acumula a soma
        quantidade = quantidade + 1  // Conta quantos pares
      }
    }
    
    // Calcular média
    media = soma / quantidade
    
    // Mostrar resultados
    escreva("Soma dos pares: ", soma, "\n")
    escreva("Quantidade de pares: ", quantidade, "\n")
    escreva("Média: ", media)
  }
}
