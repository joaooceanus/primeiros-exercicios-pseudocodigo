
//Elabore um algoritmo que mostre os números pares entre 1 e 20, inclusive. Apresente a representação em Pseudocódigo.

programa {
  funcao inicio() {
    
    inteiro numero
    
    escreva("Números pares entre 1 e 20: ")
    
    para(numero = 1; numero <= 20; numero++){
      
      se (numero % 2 == 0){
        escreva(numero, " ")
      }
    }
  }
}
